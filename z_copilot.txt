SendATCommand:
    LD HL, Command
.loop:
    LD A, (HL)
    CP 0
    RET Z
    OUT (UART_TX_PORT), A
    INC HL
    JP .loop

WaitResponse:
    IN A, (UART_TX_PORT)
    BIT 0, A
    JP Z, WaitResponse
    IN A, (UART_RX_PORT)
    ; You can store or print A here
    RET
}
